score = float(input("학점을 입력하세요: "))
if score == 4.5:
    print("입력한 학점: {}, 당신은 신입니다.".format(score))
elif 4.2<=score<4.5:
    print("입력한 학점: {}, 당신은 교수님의 사랑입니다.".format(score))
elif 3.5<=score<4.2:
    print("입력한 학점: {}, 당신은 현 체제의 수호자입니다.".format(score))
elif 2.8<=score<3.5:
    print("입력한 학점: {}, 당신은 일반인입니다.".format(score))
elif 2.3<=score<2.8:
    print("입력한 학점: {}, 당신은 일탈을 꿈꾸는 소시민입니다.".format(score))
elif 1.75<=score<2.3:
    print("입력한 학점: {}, 당신은 오락문화의 선구자입니다.".format(score))
elif 1.0<=score<1.75:
    print("입력한 학점: {}, 당신은 불가촉천민입니다.".format(score))
elif 0.5<=score<1.0:
    print("입력한 학점: {}, 당신은 자벌레입니다.".format(score))
elif 0<score<0.5:
    print("입력한 학점: {}, 당신은 플랑크톤입니다.".format(score))
elif score == 0:
    print("입력한 학점: {}, 당신은 시대를 앞서가는 혁명의 씨앗입니다.".format(score))
else:
    print("학점을 잘못 입력하였습니다.")
